Conversational Form Widget - Embed Example

Note: This example uses the local build. Once published to jsDelivr, replace the script source with the CDN URL.

How to Embed

Copy and paste this code into your website:

<!-- React and ReactDOM from CDN --> <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script> <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script> <!-- jsxRuntime compatibility layer --> <script> window.jsxRuntime = { jsx: React.createElement, jsxs: React.createElement, Fragment: React.Fragment }; </script> <!-- Your widget from jsDelivr (replace with actual URL when published) --> <script src="https://cdn.jsdelivr.net/npm/conversational-form-widget@1.0.0/dist/index.umd.js"></script> <!-- Container for the widget --> <div id="conversational-form-widget"></div> <!-- Initialize the widget --> <script> document.addEventListener('DOMContentLoaded', function() { const { ConversationalFormWidget } = window.ConversationalFormWidget; const container = document.getElementById('conversational-form-widget'); const root = ReactDOM.createRoot(container); root.render( React.createElement(ConversationalFormWidget, { apiEndpoint: 'https://your-api-endpoint.com/api/chat', onFormSubmit: function(data) { console.log('Form submitted:', data); // Handle form submission }, onFormError: function(error) { console.error('Form error:', error); // Handle errors } }) ); }); </script>
